# 实时索引

# 实时索引开关

// 在导入数据时，打开实时索引开关即可

// 实例化引擎
$Wind = new \WindSearch\Index\Wind($indexName); 
// 构建索引前，进行初始化
$Wind->buildIndexInit();
// 开启实时索引（关键步骤）
$Wind->onRealTimeIndex();

//下面进行正常的数据导入即可...
//从数据库中查询数据
foreach ($result as $v) {
    $Wind->indexer($v);
}
// 写入文件
$Wind->batchWrite();

//接下来无需构建索引,也能搜索到刚刚导入的数据
// 搜索操作...

# 注意事项

“实时索引”是顺带开启的，数据额外存储，跟增量数据不冲突，可以同时开启增量数据。 注意，不要长时间使用实时索引，不然会出现性能问题，最好每隔一段时间，就更新增量索引或全量索引。
